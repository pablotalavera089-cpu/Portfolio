/* VIDEO REEL */
let vids=[];
function parseURL(url){const yt=url.match(/(?:youtube\.com\/(?:watch\?v=|embed\/)|youtu\.be\/)([a-zA-Z0-9_-]{11})/);if(yt)return{embed:`https://www.youtube.com/embed/${yt[1]}?rel=0`,thumb:`https://img.youtube.com/vi/${yt[1]}/mqdefault.jpg`,type:'YouTube',id:yt[1]};const vi=url.match(/vimeo\.com\/(\d+)/);if(vi)return{embed:`https://player.vimeo.com/video/${vi[1]}`,thumb:null,type:'Vimeo',id:vi[1]};return null;}
function setFeat(url){document.getElementById('vfeat').innerHTML=`<iframe src="${url}" allowfullscreen allow="autoplay;fullscreen;picture-in-picture"></iframe>`;}
function promptFeat(){const u=prompt('Paste YouTube or Vimeo URL:');if(!u)return;const p=parseURL(u.trim());if(p)setFeat(p.embed);else alert('Invalid URL.');}
function addVid(){const url=document.getElementById('vurl').value.trim();const title=document.getElementById('vtitle').value.trim()||'Untitled';if(!url)return;const p=parseURL(url);if(!p){alert('Use YouTube or Vimeo link.');return;}vids.push({...p,title});document.getElementById('vurl').value='';document.getElementById('vtitle').value='';if(vids.length===1)setFeat(p.embed);renderReel();}
function delVid(i){vids.splice(i,1);renderReel();}
function renderReel(){const g=document.getElementById('rgrid');document.getElementById('vcnt').textContent=vids.length+(vids.length===1?' video':' videos');if(!vids.length){g.innerHTML='<div style="grid-column:1/-1;padding:2.8rem 0;text-align:center;"><p style="font-family:\'DM Mono\',monospace;font-size:.65rem;color:rgba(237,232,222,.2);">Add videos above</p></div>';return;}g.innerHTML=vids.map((v,i)=>`<div class="ri-card" onclick="playVid(${i})"><div class="ri-thumb">${v.thumb?`<img src="${v.thumb}" alt="${v.title}"/>`:' <div class="ri-thumb-ph"><span>▶</span></div>'}</div><div class="ri-info"><div class="ri-ttl">${v.title}</div><div class="ri-meta">${v.type}</div></div><button class="ri-del" onclick="event.stopPropagation();delVid(${i})">✕</button></div>`).join('');}
function playVid(i){setFeat(vids[i].embed);document.querySelector('.reel-feat').scrollIntoView({behavior:'smooth'});}
