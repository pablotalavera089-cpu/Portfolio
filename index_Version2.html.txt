// PAGE NAVIGATION - SINGLE FUNCTION
function gotoPage(id) {
    // Hide all pages
    document.querySelectorAll('.pg').forEach(p => p.classList.remove('on'));
    document.querySelectorAll('.pb').forEach(b => {
        b.classList.remove('on');
        b.setAttribute('aria-selected', 'false');
    });

    // Show selected page
    const page = document.getElementById('pg-' + id);
    if (page) {
        page.classList.add('on');
    }

    // Update pill nav
    const map = { home: 0, workshops: 1, performance: 2, gallery: 3, linktree: 4 };
    if (map[id] !== undefined) {
        document.querySelectorAll('.pb')[map[id]].classList.add('on');
        document.querySelectorAll('.pb')[map[id]].setAttribute('aria-selected', 'true');
    }

    window.scrollTo({ top: 0, behavior: 'smooth' });
    document.getElementById('mobileNav').classList.remove('open');
}

    document.querySelectorAll('.rv').forEach(el => io.observe(el));
}

    const grid = document.getElementById('gallery-grid');
    grid.innerHTML = galleryImages.map((img, i) => `

            <img src="${img.url}" alt="${img.label}" loading="lazy"/>

    `).join('');

    // Render admin gallery
    const adminList = document.getElementById('gallery-admin-list');
    if (galleryImages.length === 0) {
        adminList.innerHTML = '<p style="text-align:center;color:var(--mid);padding:2rem;font-size:.85rem;">No images yet. Add some photos!</p>';
    } else {
        adminList.innerHTML = galleryImages.map((img, i) => `
            <div class="gallery-item-admin">
                <img src="${img.url}" alt="${img.label}"/>
                <div class="gallery-item-admin-info">
                    <div class="gallery-item-admin-title">${img.label}</div>
                    <div class="gallery-item-admin-url">${img.url.substring(0, 50)}...</div>
                </div>
                <button class="gallery-item-admin-delete" onclick="deleteGalleryImage(${i})"><i class="fas fa-trash"></i> Delete</button>
            </div>
        `).join('');
    }
}

function updateAboutImage(url) {
    document.getElementById('about-img').src = url;
}

function addGalleryImageDirect() {
    const url = document.getElementById('direct-url').value.trim();
    const label = document.getElementById('img-label').value.trim() || 'Photo';

    if (!url) return alert('Please enter image URL');
    if (!url.includes('cloudinary')) return alert('Please use Cloudinary URLs (res.cloudinary.com/dol2caucl/...)');

    galleryImages.push({ url, label });
    localStorage.setItem('portfolio_gallery', JSON.stringify(galleryImages));
    renderGallery();
    document.getElementById('direct-url').value = '';
    document.getElementById('img-label').value = '';
    alert('✓ Image added!');
}

function deleteGalleryImage(i) {
    if (confirm('Delete this image?')) {
        galleryImages.splice(i, 1);
        localStorage.setItem('portfolio_gallery', JSON.stringify(galleryImages));
        renderGallery();
    }
}

function launchCloudinaryWidget() {
    // This uses Cloudinary's Upload Widget
    cloudinary.openUploadWidget(
        {
            cloudName: CLOUDINARY_CLOUD,
            uploadPreset: CLOUDINARY_PRESET,
            multiple: false,
            sources: ['local', 'url', 'camera']
        },
        (error, result) => {
            if (!error && result && result.event === 'success') {
                const imageUrl = result.info.secure_url;
                const label = prompt('Image label:', 'Photo');
                if (label) {
                    galleryImages.push({ url: imageUrl, label });
                    localStorage.setItem('portfolio_gallery', JSON.stringify(galleryImages));
                    renderGallery();
                    alert('✓ Image uploaded!');
                }
            }
        }
    );
}

function addEvent() {
    const day = document.getElementById('evt-day').value;
    const month = document.getElementById('evt-month').value;
    const name = document.getElementById('evt-name').value;
    const location = document.getElementById('evt-location').value;

    if (!day || !month || !name || !location) return alert('Fill all fields');

    const html = `
        <div class="sched-row">
            <div><div class="sched-day">${day}</div><div class="sched-month">${month}</div></div>
            <div><div class="sched-event">${name}</div><div class="sched-loc"><i class="fas fa-map-marker"></i> ${location}</div></div>
            <span class="sched-badge"><i class="fas fa-star"></i> Event</span>
        </div>
    `;

    document.getElementById('sched-list').innerHTML += html;
    ['evt-day', 'evt-month', 'evt-name', 'evt-location'].forEach(id => document.getElementById(id).value = '');
}
