// CURSOR

const curEl = document.getElementById('cur');
const curREl = document.getElementById('cur-r');
let mx = 0, my = 0, rx = 0, ry = 0;

document.addEventListener('mousemove', e => {
    mx = e.clientX;
    my = e.clientY;
    curEl.style.left = mx + 'px';
    curEl.style.top = my + 'px';
});

(function cursorLoop() {
    rx += (mx - rx) * .11;
    ry += (my - ry) * .11;
    curREl.style.left = rx + 'px';
    curREl.style.top = ry + 'px';
    requestAnimationFrame(cursorLoop);
})();

function curBig() {
    curEl.classList.add('big');
    curREl.classList.add('big');
}

function curSmall() {
    curEl.classList.remove('big');
    curREl.classList.remove('big');
}

function attachCursor() {
    document.querySelectorAll('a, button, .wcard, .pc, .link-card, .carousel-btn').forEach(el => {
        el.addEventListener('mouseenter', curBig);
        el.addEventListener('mouseleave', curSmall);
    });
}

function gotoSection(id) {
    // Hide all pages
    document.querySelectorAll('.pg').forEach(p => p.classList.remove('on'));
    document.querySelectorAll('.pb').forEach(b => {
        b.classList.remove('on');
        b.setAttribute('aria-selected', 'false');
    });

    // Show selected page
    const page = document.getElementById('pg-' + id);
    if (page) {
        page.classList.add('on');
        const map = { home: 0, workshops: 1, performance: 2, gallery: 3, linktree: 4 };
        if (map[id] !== undefined) {
            const btn = document.querySelectorAll('.pb')[map[id]];
            if (btn) {
                btn.classList.add('on');
                btn.setAttribute('aria-selected', 'true');
            }
        }
    }

    window.scrollTo({ top: 0, behavior: 'smooth' });
    closeMobileMenu();

    // Trigger animations
    setTimeout(() => {
        observeElements();
        attachCursor();
    }, 120);
}

// ESC key closes mobile menu and modals
document.addEventListener('keydown', e => {
    if (e.key === 'Escape') {
        closeMobileMenu();
        closeWorkshopModal();
    }
});

function initCarousel() {
    const inner = document.getElementById('carouselInner');
    const dots = document.getElementById('carouselDots');

    if (!inner || !dots) return;

    TESTIMONIALS.forEach((t, i) => {
        const item = document.createElement('div');
        item.className = 'carousel-item';
        item.innerHTML = `
            <div class="testimonial-card">
                <div class="testimonial-stars">
                    ${Array(t.rating).fill().map(() => '<i class="fas fa-star"></i>').join('')}
                </div>
                <div class="testimonial-quote">"${t.text}"</div>
                <div class="testimonial-author">
                    <div class="testimonial-avatar">${t.avatar}</div>
                    <div class="testimonial-info">
                        <h4>${t.author}</h4>
                        <p>${t.role}</p>
                    </div>
                </div>
            </div>
        `;
        inner.appendChild(item);

        const dot = document.createElement('div');
        dot.className = `dot ${i === 0 ? 'active' : ''}`;
        dot.onclick = () => showTestimonial(i);
        dots.appendChild(dot);
    });
}

function showTestimonial(n) {
    currentTestimonial = (n + TESTIMONIALS.length) % TESTIMONIALS.length;
    const inner = document.getElementById('carouselInner');
    if (inner) inner.style.transform = `translateX(-${currentTestimonial * 100}%)`;

    document.querySelectorAll('.dot').forEach((d, i) => {
        d.classList.toggle('active', i === currentTestimonial);
    });
}

function nextTestimonial() {
    showTestimonial(currentTestimonial + 1);
}

function prevTestimonial() {
    showTestimonial(currentTestimonial - 1);
}

// Auto-rotate testimonials
setInterval(nextTestimonial, 8000);

function submitBookingForm(event) {
    event.preventDefault();

    const form = document.getElementById('bookingForm');
    if (!form) return;

    const formData = new FormData(form);
    const emailContent = `
Booking Request from ${formData.get('name')}

Event: ${formData.get('eventName')}
Date: ${formData.get('eventDate')}
Location: ${formData.get('location')}

Performance Type: ${formData.get('performanceType')}
Budget: ${formData.get('budget') || 'Not specified'}

Contact:
Email: ${formData.get('email')}
Phone: ${formData.get('phone') || 'Not provided'}

Details:
${formData.get('details')}
    `;

    // Store booking locally
    let bookings = JSON.parse(localStorage.getItem('bookings')) || [];
    bookings.push({
        name: formData.get('name'),
        email: formData.get('email'),
        eventName: formData.get('eventName'),
        timestamp: new Date().toISOString()
    });
    localStorage.setItem('bookings', JSON.stringify(bookings));

    // Show success message
    form.style.display = 'none';
    const success = document.getElementById('formSuccess');
    if (success) success.classList.add('show');

    // Reset form after 3 seconds
    setTimeout(() => {
        form.reset();
        form.style.display = 'grid';
        success.classList.remove('show');
    }, 3000);
}

    grid.innerHTML = gallery.map((img, i) => `
        <div class="gallery-item rv d${(i % 3) + 1}" role="listitem">
            <img src="${img.url}" alt="${img.label}" loading="lazy"/>
        </div>
    `).join('');

    observeElements();
}

function renderLinktree() {
    const grid = document.getElementById('linktree-grid');
    if (!grid) return;

    grid.innerHTML = LINKTREE.map((link, i) => `
        <a href="${link.url}" target="_blank" rel="noopener noreferrer" class="link-card rv d${(i % 5) + 1}" role="listitem">
            <div class="link-ico"><i class="${link.ico}"></i></div>
            <div class="link-content">
                <h3>${link.title}</h3>
                <p>${link.desc}</p>
            </div>
            <div class="link-arrow"><i class="fas fa-arrow-right"></i></div>
        </a>
    `).join('');

    observeElements();
    attachCursor();
}

// Smooth scroll for anchor links
document.addEventListener('click', e => {
    if (e.target.tagName === 'A' && e.target.hash) {
        const target = document.querySelector(e.target.hash);
        if (target) {
            e.preventDefault();
            target.scrollIntoView({ behavior: 'smooth' });
        }
    }
});
